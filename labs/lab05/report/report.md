• ввести строку с клавиатуры;

• вывести введённую строку на экран.

SECTION .data

msg: DB 'Введите строку:',10 

msgLen: EQU $-msg

buf1: RESB 80

SECTION .text

GLOBAL _start

_start:

mov eax,4

mov ebx,1

mov ecx,msg

mov edx,msgLen

int 80h

mov eax, 3

mov ebx, 0

mov ecx, buf1

mov edx, 80

int 80h
